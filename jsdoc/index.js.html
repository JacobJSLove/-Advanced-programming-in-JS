		},
		error( msg ) {
			console.log ( chalk.red.bold( 'Error:' ), msg );
		},
		warrning( msg ) {
			console.log ( chalk.yellow.bold( 'Warrning:' ), msg );
		},
		info( msg ) {
			console.log ( chalk.blue.bold( 'Info:' ), msg );
		}
	};
